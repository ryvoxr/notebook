            a = -\omega^2x \\
            -\omega = \sqrt{\frac{a}{x}} \\
            \omega = \sqrt{\frac{g}{l}} = \frac{2\pi}{T} \\
            T = 2\pi\sqrt{\frac{l}{g}} \\
        \end{gather}
